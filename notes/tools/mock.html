Reqres - A hosted REST-API ready to respond to your AJAX requests (opens new window)

示例

const fetchUsers = async () => {
  try {
    const res = await fetch("https://reqres.in/api/users/");
    const json = await res.json();
    console.log(json.data);
  } catch (err) {
    console.error(err);
  }
};
